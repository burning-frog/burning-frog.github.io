iOS中使用rust

1.首先添加对应的architectures

   rustup target add aarch64-apple-ios armv7-apple-ios armv7s-apple-ios x86_64-apple-ios i386-apple-ios

2.创建项目

   mkdir rust-ios-example
   cd rust-ios-example
   
   cargo new rust --lib
   cd rust

3.编写rust

   use std::os::raw::{c_char};
   use std::ffi::{CString, CStr};
   
   #[no_mangle]
   pub extern fn rust_hello(to: *const c_char) -> *mut c_char 
     let c_str = unsafe{ CStr::from_ptr(to) }
     let recipient = match c_str.to_str() {
       Err(_) => "there",
       Ok(string) => string,
     };
   	CString::new("hello".to_owned() + recipient).unwarp().into_raw()
   }
   
   #[no_mangle]
   pub extern fn rust_hello_free(s: *mut c_char) {
     unsafe {
       if s.is_null() { return }
       CString::from_raw(s)
     }
   }

4.在Cargo.toml中修改成以下设置

   cbindgen src/lib.rs -l c > rust.h

   [lib]
   name = "rust"
   crate-type = ["staticlib", "cydlib"]

5.在xcode项目中使用

   import UIKit
   class ViewController: UIViewController {
     override func viewDidLoad() {
       super.viewDidLoad()
       
       let resut = rust_hello("world")
       let swift_resut = String(cString: resutl!)
       rust_hello_free(UnsafeMutablePointer(mutating: resut))
       print(swift_result)
     }
   }

6.将头文件和.a文件加入xcode中连接

记得设置Header search Path 和Library search path以及Objective-c Bridging Header

   cd ~/rust-ios-example
   
   mkdir hello-rust/libs
   mkdir hello-rust/include
   
   cp rust/rust.h hello-rust/include
   cp rust/target/universal/release/librust.a hello-rust/libs
